# Step 1: Generate Synthetic Data
from ml.ml_testing_framework import HistoricalDataGenerator
gen = HistoricalDataGenerator(days=90)
df_bullish = gen.generate_trending_candles('BULLISH')
df_bearish = gen.generate_trending_candles('BEARISH')
df_cons = gen.generate_consolidation_candles()
df_rev = gen.generate_reversal_candles()
df_volatile = gen.generate_volatile_candles(2.0)

# Step 2: Load Real Data + Validate
from ml.ml_super_tester import MLDataLoader
loader = MLDataLoader()
df_real = loader.load_real_training_data(days=30)

# Step 3: Feature Validation
from ml.ml_super_tester import FeatureEngineValidator
validator = FeatureEngineValidator()
feature_report = validator.validate_all_features(df_real)
print(f"Feature Coverage: {feature_report['overall']['coverage_pct']}%")

# Step 4: Trigger Analysis
from ml.ml_super_tester import TriggerAnalyzer
analyzer = TriggerAnalyzer()
triggers = analyzer.analyze_trigger_correlation(df_real)
for trigger, stats in triggers['trigger_win_rates'].items():
    print(f"{trigger}: {stats['win_rate']}% WR")

# Step 5: Train Model
from ml.ml_super_tester import MLModelTrainer
from sklearn.model_selection import train_test_split
df_train, df_test = train_test_split(df_real, test_size=0.2)
trainer = MLModelTrainer()
result = trainer.train_and_validate(df_train, df_test)
print(f"Test F1: {result['model_metrics']['test']['f1']}")
print(f"Top Features: {result['top_features']}")

# Step 6: Scenario Testing
from ml.ml_super_tester import ScenarioTester
tester = ScenarioTester()
scenarios = tester.test_all_scenarios(df_real)
for scenario, stats in scenarios.items():
    print(f"{scenario}: {stats['stats']['win_rate']}% WR")
